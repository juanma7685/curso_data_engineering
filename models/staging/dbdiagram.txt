Table users {
  user_id int [pk, unique, not null]
  updated_at datetime [not null]
  address_id int
  last_name varchar(50)
  created_at datetime [not null]
  phone_number varchar
  first_name varchar(50)
  email varchar
  llegada_id int [not null]
}

Table status {
  status_id int [pk, unique, not null]
  status varchar
}

Table shipping_service {
  shipping_service_id int [pk, unique, not null]
  shipping_service varchar
}

Table promos {
  promo_id int [pk, unique, not null]
  name varchar
  EUR_discount decimal
  status varchar
  llegada_id int
}

Table products {
  product_id int [pk, unique, not null]
  precio decimal
  nombre_producto varchar
  category_id int
  inventory int
  llegada_id int
}

Table orders {
  order_id int [pk, unique, not null]
  address_id int
  user_id int
  shipping_cost decimal
  created_at datetime [not null]
  estimated_delivery_at datetime
  order_cost decimal
  order_total decimal
  delivered_at datetime
  tracking_id varchar
  shipping_service_id int
  promo_id int
  status_id int
  llegada_id int [not null]
}

Table order_items {
  order_items_id int [pk, unique, not null]
  order_id int
  product_id int
  quantity int
  llegada_id int
}

Table events {
  event_id int [pk, unique, not null]
  page_url varchar
  domain varchar
  page_id int
  event_type_id int
  user_id int
  product_id int
  session_id int [not null]
  created_at datetime [not null]
  llegada_id int [not null]
}

Table event_type {
  event_type_id int [pk, unique, not null]
  event_type varchar
}

Table addresses {
  address_id int [pk, unique, not null]
  zipcode varchar
  country varchar
  address varchar
  address_number varchar
  street varchar
  state varchar
  llegada_id int
}

Table category {
  category_id int [pk, unique, not null]
  category varchar
}

Table snap_product {
  product_id int [pk, unique, not null]
  precio decimal
  nombre_producto varchar
  category_id int
  inventory int
  llegada_id int
}

Table budget {
  budget_id int [pk, unique, not null]
  product_id int
  quantity int
  mes int
  anyo int
  llegada_id int [not null]
}

Table reviews {
  review_id int [pk, unique, not null]
  user_id int
  product_id int
  rating int
  comments varchar
  created_at date
  llegada_id int [not null]
}

Table returns {
  return_id int [pk, unique, not null]
  order_id int
  user_id int
  return_date datetime
  llegada_id int [not null]
}

Table snap_reviews {
  review_id int
  user_id int
  product_id int
  rating int
  comments varchar
  created_at date
  llegada_id int
}

// Relationships
Ref: users.address_id > addresses.address_id
Ref: orders.address_id > addresses.address_id
Ref: orders.user_id > users.user_id
Ref: orders.shipping_service_id > shipping_service.shipping_service_id
Ref: orders.promo_id > promos.promo_id
Ref: orders.status_id > status.status_id
Ref: order_items.order_id > orders.order_id
Ref: order_items.product_id > products.product_id
Ref: events.event_type_id > event_type.event_type_id
Ref: events.user_id > users.user_id
Ref: events.product_id > products.product_id
Ref: budget.product_id > products.product_id
Ref: reviews.user_id > users.user_id
Ref: reviews.product_id > products.product_id
Ref: returns.order_id > orders.order_id
Ref: returns.user_id > users.user_id
